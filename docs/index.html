üöÄ Quick Integration

Add BlockVerify to your website with just one script tag:

<!-- Latest version from jsDelivr CDN -->
<script src="https://cdn.jsdelivr.net/gh/yourusername/blockverify@main/client_sdk/blockverify.min.js"></script>

<!-- Initialize with your API key -->
<script>
  BlockVerify.init({
    apiKey: 'bv_prod_your_key_here',
    minAge: 18,
    onSuccess: (result) => console.log('User verified', result),
    onFailure: (error) => console.log('Verification failed', error)
  });
</script>
